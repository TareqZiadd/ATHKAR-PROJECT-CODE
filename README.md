    const style3={fontSize:"20px",color:"grey",fontFamily:"revert"}

    const [isVisible, setIsVisible] = useState(true);

        if (countIncr === 0) {
            setIsVisible(false);

        if (countIncr > 0) {
            setCountIncr(countIncr - 1);
        }

            )}
            {isVisible && (
                <><button onClick={handleClick} type="button" className="btn btn-danger" style={style2}>
                    0{countIncr}
                </button>
</>  

            )}

        </div>
    );
}
